  // This widget is the home page of your application. It is stateful, meaning

  final String title;

  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  final List<PlayerInfo> playerInfo = List.empty(growable: true);
  String serverIpAddr = "192.168.1.120";
  String playerIpAddr = "";

  // Create a text controller and use it to retrieve the current value
  // of the TextField.
  final ipAddrController = TextEditingController();

  String log = "";
  final logViewController = TextEditingController();

  _MyHomePageState() {
    ipAddrController.text = serverIpAddr;
    logViewController.text = log;
    _serverQuery();
  }

  @override
  void dispose() {
    // Clean up the controller when the widget is disposed.
    ipAddrController.dispose();
    logViewController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    // This method is rerun every time setState is called, for instance as done
    // by the _incrementCounter method above.

      appBar: AppBar(

        // the App.build method, and use it to set our appbar title.
        title: Text(widget.title),
      ),
      body: Container(

                ),
                const SizedBox(width: 10),
                ElevatedButton(
                  onPressed: () {

                          _rebootPlayer();

                          textScaleFactor: 1.25,

              )

  late Socket socket;
  bool connectionIsOpen = false;

  void _serverQuery() async {
    if (connectionIsOpen) {
      append('request to connect to LMS $serverIpAddr declined due to open connection');
      return;
    }

    log = ""; // reset log when button is pressed
    playerInfo.clear();
    var addr = InternetAddress(serverIpAddr);
    Socket.connect(addr, 9090).then((Socket sock) {
      socket = sock;
      append("connected to LMS ${socket.remoteAddress.address}:${socket.remotePort}");
      connectionIsOpen = true;
      socket.listen(serverDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
      socket.write('player count ?\n');
    }).catchError((AsyncError e) {
        append("Unable to connect to LMS, error caught: $e");
    });
  }

  int playerCount = 0;
  int currentPlayerIdx = -1;
  String playerModelQueryStr = "";
  String playerNameQueryStr = "";
  String playerIpQueryStr = "";
  String currentPlayerName = "";
  
  void serverDataHandler(data) {
    if (!connectionIsOpen) {
      return;
    }

    /*
        player count ?
        (iterate over returned count: player count x)
        player model idx ?
        (player model idx fab4)
        player name idx ?
        (player name idx foo)
        player ip idx ?
        (player ip idx addr:port)
    */
    var txt = String.fromCharCodes(data).trim();
    if (txt.startsWith("player count ")) {
      txt = txt.substring(13);
      playerCount = int.parse(txt);
      currentPlayerIdx = 0;
      serverQueryNextPlayer();
    }
    else if (txt.startsWith(playerModelQueryStr)) {
      // append("[server] $txt");
      if (txt.contains(" fab4")) {
        socket.write(playerNameQueryStr + " ?\n");
      }
      else {
        serverQueryNextPlayer();
      }
    }
    else if (txt.startsWith(playerNameQueryStr)) {
      txt = unescapeText(txt);
      currentPlayerName = txt.substring(playerNameQueryStr.length + 1);
      socket.write(playerIpQueryStr + " ?\n");
    }
    else if (txt.startsWith(playerIpQueryStr)) {
      PlayerInfo inf = PlayerInfo();
      txt = unescapeText(txt);
      int pos = txt.indexOf(':');
      if (pos == -1) {
        inf.pIpAddr = txt.substring(playerIpQueryStr.length + 1);
      }
      else {
        inf.pIpAddr = txt.substring(playerIpQueryStr.length + 1, pos);
      }
      inf.pName = currentPlayerName;
      playerInfo.add(inf);

      serverQueryNextPlayer();
    }
    else {
      append("[unhandled server response] $txt");
    }
  }

  String unescapeText(String txt) {
      txt = txt.replaceAll("%3A", ":");
      txt = txt.replaceAll("%20", " ");
      return txt;
  }

  void serverQueryNextPlayer() {
    if (currentPlayerIdx < playerCount) {
      playerModelQueryStr = "player model $currentPlayerIdx";
      playerNameQueryStr = "player name $currentPlayerIdx";
      playerIpQueryStr = "player ip $currentPlayerIdx";
      socket.write(playerModelQueryStr + " ?\n");
      currentPlayerIdx++;
    }
    else {
      socket.write('exit\n');
      connectionIsOpen = false;
      socket.close();
      append("server connection closed");
      setState(() { });
    }
  }

  void _rebootPlayer() async {
    if (connectionIsOpen) {
      append('request to reboot $playerIpAddr declined due to open connection');
      return;
    }

    log = ""; // reset log when button is pressed
    append('attempting to reboot $playerIpAddr');
    var addr = InternetAddress(playerIpAddr);
    Socket.connect(addr, 23).then((Socket sock) {
      socket = sock;
      append("connected to ${socket.remoteAddress.address}:${socket.remotePort}");
      connectionIsOpen = true;
      socket.listen(playerDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
    }).catchError((AsyncError e) {
        append("Unable to connect to player, error caught: $e");
    });
  }

  void playerDataHandler(data) {
    if (!connectionIsOpen) {
      return;
    }

    Uint8List telnetNegotiation = Uint8List.fromList([255, 253, 1, 255, 253, 31, 255, 251, 1, 255, 251, 3]);
    var txt = String.fromCharCodes(data).trim();
    if (data.length == telnetNegotiation.length && txt == String.fromCharCodes(telnetNegotiation)) {
      // telnet negotiation settings in initial response, 
      // sometimes with login prompt, sometimes alone
      append("player sent telnet negotiation");
    }
    else if (txt.endsWith("SqueezeboxTouch login:")) {
      if (txt.length > 25) {
        txt = txt.substring(15);
        append("[player responded (with telnet negotiation)] $txt");
      } 
      else {
        append("[player responded] $txt");
      }
      // login name: root
      socket.write('root\n');
    }
    else if (txt.endsWith("Password:")) {
      append("[player responded] $txt");
      // login password: 1234
      socket.write('1234\n');
    }
    else if (txt.contains("Enjoy!")) {
      append("logged in, sending reboot command");
      // command: reboot
      socket.write('reboot\n');
      // socket.write('exit\n'); // for debugging
      sleep(const Duration(milliseconds: 500)); // reboot fails if socket is closed too quickly
      connectionIsOpen = false;
      append("closing player connection");
      socket.close();
    }
    else if (
      txt == "root" || 
      txt == "") {
      // don't log anything
    }
    else {
      append("[unhandled player response] $txt");
    }
  }

  void socketErrorHandler(error, StackTrace trace) {
    if (!connectionIsOpen) {
      return;
    }

    append("[error] $error");
  }

  void socketDoneHandler() {
    connectionIsOpen = false;
    socket.destroy();
  }

  void append(text) {
    log += "\n$text";
    logViewController.text = log;
  }
}
