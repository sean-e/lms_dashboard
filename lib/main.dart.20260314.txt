late SharedPreferences prefs; 

void main() async {
  // ensureInitialized is moved from _DashboardAppState.init because it is 
  // required for SharedPreferences use below
  WidgetsFlutterBinding.ensureInitialized();

  // await can't happen in _DashboardAppState.init because it doesn't complete 
  // before _DashboardHomePageState is constructed where the prefs are needed
  prefs = await SharedPreferences.getInstance();

  runApp(const DashboardApp());
}

  String playerId = "";
  String deviceName = "";
  String wifiSignalStrength = "";
  int powerState = 0;
  bool supportsTelnet = false;
}

    ipAddrController.text = serverIpAddr;
    ipAddrPortController.text = "$serverIpAddrPort";
    logViewController.text = log;

    _serverQuery();
  }

  void submitQuery() {
    serverIpAddr = ipAddrController.text;
    serverIpAddrPort = int.parse(ipAddrPortController.text);

    _serverQuery();

    prefs.setString("server_ip_addr", serverIpAddr);
    prefs.setInt("server_ip_addr_port", serverIpAddrPort);
    prefs.setString("player_telnet_username", playerTelnetUserName);
    prefs.setString("player_telnet_password", playerTelnetPassword);
  }

  late Socket socket;
  bool connectionIsOpen = false;

  void _serverQuery() async {
    if (connectionIsOpen) {
      logActivity("request to connect to LMS $serverIpAddr declined due to open connection");
      return;
    }

    if (serverIpAddr.isEmpty || serverIpAddrPort == 0) {
      return;
    }

    log = "Activity:"; // reset log
    logActivity("opening connection to $serverIpAddr:$serverIpAddrPort");
    playerInfo.clear();
    InternetAddress addr;
    try {
      addr = InternetAddress(serverIpAddr);
    }
    catch(e) {
      logActivity("$e");
      return;
    }

    Duration timeoutVal = const Duration(seconds:10);
    Socket.connect(addr, serverIpAddrPort, timeout:timeoutVal).then((Socket sock) {
      socket = sock;
      logActivity("connected to LMS, begin server query");
      connectionIsOpen = true;
      socket.listen(serverQueryDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
      socket.write("player count ?\n");
    }).catchError((e) {
      logActivity("Unable to connect to LMS, error caught: $e");
      setState(() {});
    });
  }

  int playerCount = 0;
  int queryPlayerIdx = -1;
  String playerConnectedQueryStr = "";
  String playerModelQueryStr = "";
  String playerNameQueryStr = "";
  String playerIpQueryStr = "";
  String playerIdQueryStr = "";
  String playerWifiQueryStr = "";
  String playerPowerQueryStr = "";
  PlayerInfo currentQueryPlayer = PlayerInfo();
  
  void serverQueryDataHandler(Uint8List data) {
    if (!connectionIsOpen) {
      return;
    }

    /*
        player count ?
        (iterate over returned count: player count x)
        player model idx ?
        (player model idx fab4)
        player name idx ?
        (player name idx foo)
        player ip idx ?
        (player ip idx addr:port)
    */
    var txt = String.fromCharCodes(data).trim();
    txt = unescapeText(txt);
    if (txt.startsWith("player count ")) {
      txt = txt.substring(13);
      playerCount = int.parse(txt);
      queryPlayerIdx = 0;
      serverQueryNextPlayer();
    }
    else if (txt.startsWith(playerIdQueryStr)) {
      currentQueryPlayer.playerId = txt.substring(playerIdQueryStr.length);

      playerConnectedQueryStr = "${currentQueryPlayer.playerId} connected ";
      playerWifiQueryStr = "${currentQueryPlayer.playerId} signalstrength ";
      playerPowerQueryStr = "${currentQueryPlayer.playerId} power ";

      socket.write("$playerConnectedQueryStr?\n");
    }
    else if (txt.startsWith(playerConnectedQueryStr)) {
      int connected = int.parse(txt.substring(playerConnectedQueryStr.length));
      if (connected == 1) {
        socket.write("$playerModelQueryStr?\n");
      }
      else {
        // don't display disconnected players
        serverQueryNextPlayer();
      }
    }
    else if (txt.startsWith(playerModelQueryStr)) {
      if (txt.contains(" fab4")) {
        currentQueryPlayer.supportsTelnet = true;
      }
      else {
        currentQueryPlayer.supportsTelnet = false;
      }
      socket.write("$playerNameQueryStr?\n");
    }
    else if (txt.startsWith(playerNameQueryStr)) {
      currentQueryPlayer.deviceName = txt.substring(playerNameQueryStr.length);
      socket.write("$playerIpQueryStr?\n");
    }
    else if (txt.startsWith(playerIpQueryStr)) {
      int pos = txt.indexOf(':');
      if (pos == -1) {
        currentQueryPlayer.ipAddr = txt.substring(playerIpQueryStr.length);
      }
      else {
        currentQueryPlayer.ipAddr = txt.substring(playerIpQueryStr.length, pos);
      }
      socket.write("$playerWifiQueryStr?\n");
    }
    else if (txt.startsWith(playerWifiQueryStr)) {
      currentQueryPlayer.wifiSignalStrength = txt.substring(playerWifiQueryStr.length);
      socket.write("$playerPowerQueryStr?\n");
    }
    else if (txt.startsWith(playerPowerQueryStr)) {
      txt = txt.substring(playerPowerQueryStr.length);
      currentQueryPlayer.powerState = int.parse(txt);
      serverQueryNextPlayer();
    }
    else {
      logActivity("[unhandled server response] $txt");
    }
  }

  String unescapeText(String txt) {
      txt = txt.replaceAll("%3A", ":");
      txt = txt.replaceAll("%3F", "?");
      txt = txt.replaceAll("%20", " ");
      return txt;
  }

  void serverQueryNextPlayer() {
    if (currentQueryPlayer.deviceName.isNotEmpty) {
      playerInfo.add(currentQueryPlayer);
      currentQueryPlayer = PlayerInfo();
    }

    if (queryPlayerIdx < playerCount) {
      playerModelQueryStr = "player model $queryPlayerIdx ";
      playerNameQueryStr = "player name $queryPlayerIdx ";
      playerIpQueryStr = "player ip $queryPlayerIdx ";
      playerIdQueryStr = "player id $queryPlayerIdx ";
      logActivity("player ${queryPlayerIdx+1}");
      socket.write("$playerIdQueryStr?\n");
      queryPlayerIdx++;
    }
    else {
      socket.write("exit\n");
      connectionIsOpen = false;
      socket.close();
      logActivity("server connection closed");
      setState(() {});
    }
  }

  void _rebootPlayer() async {
    if (connectionIsOpen) {
      logActivity("request to reboot ${selectedPlayer.deviceName} declined due to open connection");
      return;
    }

    log = "Activity:"; // reset log when button is pressed
    logActivity("attempting to reboot ${selectedPlayer.deviceName}");
    InternetAddress addr;
    try {
      addr = InternetAddress(selectedPlayer.ipAddr);
    }
    catch(e) {
      logActivity("$e");
      return;
    }

    Duration timeoutVal = const Duration(seconds:10);
    Socket.connect(addr, 23, timeout:timeoutVal).then((Socket sock) {
      socket = sock;
      logActivity("connected to ${socket.remoteAddress.address}:${socket.remotePort}");
      connectionIsOpen = true;
      socket.listen(playerDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
    }).catchError((e) {
      logActivity("Unable to connect to player, error caught: $e");
    });
  }

  void playerDataHandler(Uint8List data) {
    if (!connectionIsOpen) {
      return;
    }

    Uint8List telnetNegotiation = Uint8List.fromList([255, 253, 1, 255, 253, 31, 255, 251, 1, 255, 251, 3]);
    var txt = String.fromCharCodes(data).trim();
    if (data.length == telnetNegotiation.length && txt == String.fromCharCodes(telnetNegotiation)) {
      // telnet negotiation settings in initial response, 
      // sometimes with login prompt, sometimes alone
      logActivity("player sent telnet negotiation");
    }
    else if (txt.endsWith("SqueezeboxTouch login:")) {
      if (txt.length > 25) {
        txt = txt.substring(15);
        logActivity("[player responded (with telnet negotiation)] $txt");
      } 
      else {
        logActivity("[player responded] $txt");
      }
      // login name: root
      socket.write("$playerTelnetUserName\n");
    }
    else if (txt.endsWith("Password:")) {
      logActivity("[player responded] $txt");
      // login password: 1234
      socket.write("$playerTelnetPassword\n");
    }
    else if (txt.contains("Enjoy!")) {
      logActivity("logged in, sending reboot command");
      // command: reboot
      socket.write("reboot\n");
      // socket.write("exit\n"); // for debugging
      sleep(const Duration(milliseconds: 500)); // reboot fails if socket is closed too quickly
      connectionIsOpen = false;
      logActivity("closing player connection");
      socket.close();
    }
    else if (
      txt == playerTelnetUserName || 
      txt == "") {
      // don't log anything
    }
    else {
      logActivity("[unhandled player response] $txt");
    }
  }

  // server player control strings
  String playerPowerOffCommandStr = "";
  String playerPowerOnCommandStr = "";
  String playerPlayCommandStr = "";
  String playerStopCommandStr = "";

  void serverPlayerControlDataHandler(Uint8List data) {
    if (!connectionIsOpen) {
      return;
    }

    var txt = String.fromCharCodes(data).trim();
    txt = unescapeText(txt);
    if (txt.startsWith(playerPowerOnCommandStr)) {
      logActivity("[server response] $txt");
      socket.write("$playerStopCommandStr\n");
    }
    else if (txt.startsWith(playerStopCommandStr)) {
      logActivity("[server response] $txt");
      socket.write("$playerPlayCommandStr\n");
    }
    else if (txt.startsWith(playerPowerOffCommandStr) || txt.startsWith(playerPlayCommandStr)) {
      logActivity("[server response] $txt");
      connectionIsOpen = false;
      socket.write("exit\n");
      socket.close();
      logActivity("server connection closed, pausing before refresh");
      Future.delayed(Duration(seconds: 2), () { 
        _serverQuery();
      });
    }
    else {
      logActivity("[unhandled server response] $txt");
    }
  }

  void _togglePlayerPower() async {
    if (selectedPlayer.powerState == 1) {
      _powerOffPlayer();
    }
    else if (selectedPlayer.powerState == 0) {
      _powerOnPlayer();
    }
  }

  void _powerOffPlayer() async {
    if (connectionIsOpen) {
      logActivity("request to power off ${selectedPlayer.deviceName} declined due to open connection");
      return;
    }

    log = "Activity:"; // reset log when button is pressed
    logActivity("attempting to power off ${selectedPlayer.deviceName}");
    InternetAddress addr;
    try {
      addr = InternetAddress(serverIpAddr);
    }
    catch(e) {
      logActivity("$e");
      return;
    }

    Duration timeoutVal = const Duration(seconds:10);
    Socket.connect(addr, serverIpAddrPort, timeout:timeoutVal).then((Socket sock) {
      socket = sock;
      logActivity("connected to LMS");
      connectionIsOpen = true;
      socket.listen(serverPlayerControlDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
      updatePlayerControlStrings();
      socket.write("$playerPowerOffCommandStr\n");
    }).catchError((e) {
      logActivity("Unable to connect to LMS, error caught: $e");
      setState(() {});
    });
  }

  void _powerOnPlayer() async {
    if (connectionIsOpen) {
      logActivity("request to power on ${selectedPlayer.deviceName} declined due to open connection");
      return;
    }

    log = "Activity:"; // reset log when button is pressed
    logActivity("attempting to power on ${selectedPlayer.deviceName}");
    InternetAddress addr;
    try {
      addr = InternetAddress(serverIpAddr);
    }
    catch(e) {
      logActivity("$e");
      return;
    }

    Duration timeoutVal = const Duration(seconds:10);
    Socket.connect(addr, serverIpAddrPort, timeout:timeoutVal).then((Socket sock) {
      socket = sock;
      logActivity("connected to LMS");
      connectionIsOpen = true;
      socket.listen(serverPlayerControlDataHandler, 
        onError: socketErrorHandler, 
        onDone: socketDoneHandler, 
        cancelOnError: true);
      updatePlayerControlStrings();
      socket.write("$playerPowerOnCommandStr\n");
    }).catchError((e) {
      logActivity("Unable to connect to LMS, error caught: $e");
      setState(() {});
    });
  }

  void updatePlayerControlStrings() {
    playerPlayCommandStr = "${selectedPlayer.playerId} button play";
    playerStopCommandStr = "${selectedPlayer.playerId} button stop";
    playerPowerOnCommandStr = "${selectedPlayer.playerId} power 1";
    playerPowerOffCommandStr = "${selectedPlayer.playerId} power 0";
  }

  void socketErrorHandler(Object error, StackTrace trace) {
    // if (!connectionIsOpen) {
    //   return;
    // }

    logActivity("[error] $error");
    socket.close();
    socketDoneHandler();
  }

  void socketDoneHandler() {
    connectionIsOpen = false;
    socket.destroy();
  }

  void logActivity(String text) {
    log += "\n$text";
    logViewController.text = log;
  }
}
